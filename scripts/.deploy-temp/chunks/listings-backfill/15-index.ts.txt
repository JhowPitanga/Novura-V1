// deno-lint-ignore-file no-explicit-any
// listings-backfill — migrates existing data from marketplace_items_raw into
// the canonical tables by running the listing adapters on each stored payload.
//
// Request body:
//   {
//     organizationId: string,
//     marketplaceName?: string,   // filter to one channel; omit for both
//     pageSize?: number,          // default 100
//     offset?: number,            // default 0 (for pagination)
//     dryRun?: boolean            // default false
//   }
//
// This function is idempotent: all writes use UPSERT, so it is safe to re-run.

declare const Deno: any;

import { jsonResponse, handleOptions } from "../_shared/adapters/infra/http-utils.ts";
import { createAdminClient } from "../_shared/adapters/infra/supabase-client.ts";
import { syncCanonicalFromPayload } from "../_shared/listing-adapters/syncCanonicalFromPayload.ts";
import { fetchMlSupplementary, isMercadoLivreChannel } from "../_shared/listing-adapters/reconcileCanonical.ts";
import type { AdapterNormalizeExtra } from "../_shared/listing-adapters/types.ts";

Deno.serve(async (req: Request) => {
  if (req.method === "OPTIONS") return handleOptions();
  if (req.method !== "POST") return jsonResponse({ error: "Method not allowed" }, 405);

  const SUPABASE_URL = Deno.env.get("SUPABASE_URL");
  const SERVICE_ROLE_KEY = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY");

  if (!SUPABASE_URL || !SERVICE_ROLE_KEY) {
    return jsonResponse({ error: "Missing service configuration" }, 500);
  }

  let body: Record<string, unknown>;
  try {
    body = await req.json();
  } catch {
    return jsonResponse({ error: "Invalid JSON body" }, 400);
  }

  const organizationId = String(body.organizationId ?? '');
  const marketplaceName = body.marketplaceName ? String(body.marketplaceName) : null;
  const pageSize = Math.min(Number(body.pageSize ?? 100), 500);
  const offset = Number(body.offset ?? 0);
  const dryRun = body.dryRun === true;

  if (!organizationId) {
    return jsonResponse({ error: "organizationId is required" }, 400);
  }

  const supabase = createAdminClient();

  // -------------------------------------------------------------------------
  // Create audit row
  // -------------------------------------------------------------------------
  const { data: jobRow } = await supabase
    .from("marketplace_listing_sync_jobs" as any)
    .insert({
      organizations_id: organizationId,
      marketplace_name: marketplaceName ?? 'all',
      marketplace_item_id: '*',
      scope: 'backfill',
      status: 'running',
      started_at: new Date().toISOString(),
    })
    .select("id")
    .single();

  const jobId: string | null = jobRow?.id ?? null;
  const startedAt = Date.now();

  const updateJob = async (status: string, errorMsg?: string, extra?: Record<string, unknown>) => {
    if (!jobId) return;
    const finishedAt = new Date();
    await supabase
      .from("marketplace_listing_sync_jobs" as any)
      .update({
        status,
        error_message: errorMsg ?? null,
        finished_at: finishedAt.toISOString(),
        duration_ms: Date.now() - startedAt,
        ...extra,
      })
      .eq("id", jobId);
  };

  try {
    // -------------------------------------------------------------------------
    // Fetch a page of raw items
    // -------------------------------------------------------------------------
    let query = supabase
      .from("marketplace_items_raw" as any)
      .select("id, marketplace_item_id, marketplace_name, marketplace, integration_id, data, updated_at")
      .eq("organizations_id", organizationId)
      .order("id", { ascending: true })
      .range(offset, offset + pageSize - 1);

    if (marketplaceName) {
      query = query.eq("marketplace_name", marketplaceName);
    }

    const { data: rawItems, error: fetchErr } = await query;

    if (fetchErr) {
      await updateJob('error', fetchErr.message);
      return jsonResponse({ error: fetchErr.message }, 500);
    }

    const items: any[] = rawItems ?? [];
    const results = { processed: 0, skipped: 0, errors: [] as string[] };

    const mlItemIds = items
      .filter((it) => isMercadoLivreChannel(String(it.marketplace_name ?? it.marketplace ?? '')))
      .map((it) => String(it.marketplace_item_id))
      .filter(Boolean);

    const metricsByItemId = new Map<string, Record<string, unknown>>();
    const pricesByItemId = new Map<string, Record<string, unknown>>();

    if (mlItemIds.length > 0) {
      const [metricsRes, pricesRes] = await Promise.all([
        supabase
          .from('marketplace_metrics')
          .select('*')
          .eq('organizations_id', organizationId)
          .eq('marketplace_name', 'Mercado Livre')
          .in('marketplace_item_id', mlItemIds),
        supabase
          .from('marketplace_item_prices')
          .select('*')
          .eq('organizations_id', organizationId)
          .eq('marketplace_name', 'Mercado Livre')
          .in('marketplace_item_id', mlItemIds),
      ]);
      for (const row of metricsRes.data ?? []) {
        metricsByItemId.set(String(row.marketplace_item_id), row);
      }
      for (const row of pricesRes.data ?? []) {
        pricesByItemId.set(String(row.marketplace_item_id), row);
      }
    }

    for (const item of items) {
      const mktName: string =
        item.marketplace_name ?? item.marketplace ?? '';

      if (!mktName) {
        results.skipped++;
        continue;
      }

      try {
        const itemId = String(item.marketplace_item_id);
        let mlExtra: AdapterNormalizeExtra | undefined;
        if (isMercadoLivreChannel(mktName)) {
          const metricsRow = metricsByItemId.get(itemId) ?? null;
          mlExtra = {
            metricsRow,
            listingPricesRow: pricesByItemId.get(itemId) ?? null,
            qualityRow: metricsRow,
          };
        }

        if (!dryRun) {
          const syncResult = await syncCanonicalFromPayload(supabase as any, {
            organizationId,
            integrationId: item.integration_id ?? null,
            marketplaceName: mktName,
            marketplaceItemId: itemId,
            payload: item,
            payloadSource: 'backfill',
            mlExtra,
          });

          if (!syncResult.ok) {
            results.errors.push(`${itemId}: ${syncResult.error ?? 'sync failed'}`);
            continue;
          }
        }

        results.processed++;
      } catch (err: any) {
        results.errors.push(`${item.marketplace_item_id}: ${err?.message ?? String(err)}`);
      }
    }

    const summary = {
      total: items.length,
      processed: results.processed,
      skipped: results.skipped,
      errors: results.errors.length,
      errorDetails: results.errors.slice(0, 20),
      dryRun,
      offset,
      pageSize,
      hasMore: items.length === pageSize,
    };

    const status = results.errors.length === 0 ? 'success' : 'success_with_errors';
    await updateJob(status, results.errors.length > 0 ? results.errors.join('; ').slice(0, 500) : undefined);

    return jsonResponse({ ok: true, jobId, ...summary });
  } catch (err: any) {
    const msg = err?.message ?? String(err);
    await updateJob('error', msg);
    return jsonResponse({ error: msg }, 500);
  }
});
